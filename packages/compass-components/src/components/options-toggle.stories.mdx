# OptionsToggle

The OptionsToggle component is a button that toggles the visibility of additional options or content. It's commonly used to show/hide advanced settings or supplementary information.

## Usage

```tsx
import { OptionsToggle } from '@mongodb-js/compass-components';

function OptionsToggleExample() {
  const [isExpanded, setIsExpanded] = React.useState(false);

  return (
    <div>
      <OptionsToggle
        isExpanded={isExpanded}
        onToggleOptions={() => setIsExpanded(!isExpanded)}
        aria-controls="options-content"
      />
      {isExpanded && (
        <div id="options-content">Additional options content goes here</div>
      )}
    </div>
  );
}
```
